1หลักการออกแบบ (Design Principles)

  • Normalization (3NF) — ออกแบบตารางให้อยู่ในรูปแบบบรรทัดฐานที่สาม ลดความซ้ำซ้อน (redundancy) และความผิดปกติของข้อมูล (anomaly) ยกเว้นกรณีจำเป็นด้านประสิทธิภาพจึง denormalize อย่างมีเหตุผล
  • Referential Integrity — บังคับความสัมพันธ์ด้วย Foreign Key Constraint ทุกความสัมพันธ์ พร้อมกำหนด ON DELETE/UPDATE ที่เหมาะสม (RESTRICT/CASCADE)
  • Surrogate Key — ทุกตารางใช้ <table>_id (BIGINT) เป็น PK เพื่อความเสถียร ส่วน Business Key (เช่น HN, VN) ใช้ UNIQUE Constraint
  • Data Type ข้ามแพลตฟอร์ม — ใช้ชนิดข้อมูลมาตรฐาน ISO SQL รองรับ PostgreSQL / SQL Server / Oracle (ค่าเงิน NUMERIC, เวลา TIMESTAMP)
  • Audit & Soft Delete — ทุกตารางหลักมี created_at, updated_at และพิจารณา is_deleted สำหรับ soft delete ข้อมูลผู้ป่วย
  • รองรับการ Migration — มีฟิลด์ legacy_hn/source_ref เพื่อ map ข้อมูลเดิมจาก HosXP และรองรับ reconciliation 100%
  • ความปลอดภัยข้อมูล — ข้อมูล PII/สุขภาพ เข้ารหัสและควบคุมการเข้าถึงตามสิทธิ์ (สอดคล้อง PDPA) · ข้อมูลผู้ป่วยเป็นกรรมสิทธิ์ของโรงพยาบาล

2แผนภาพความสัมพันธ์ของข้อมูล (ERD)

แสดงความสัมพันธ์หลักของเอนทิตี โดยใช้สัญลักษณ์ 1 ──< N หมายถึงความสัมพันธ์แบบหนึ่งต่อกลุ่ม (One-to-Many)

หมายเหตุ: patient 1:N visit, visit 1:N order_item, patient 1:N admission (IPD), visit/admission 1:N prescription, prescription 1:N prescription_item (เชื่อม drug → eMAR), users (doctor/nurse/pharmacist) 1:N visit/emar — รองรับ Smart IPD

4นิยามตารางตัวอย่าง (Table Definition — DDL)

ตัวอย่าง DDL เชิงตรรกะ (ปรับชนิดข้อมูลให้ตรง DBMS ที่ใช้จริง) — แสดง field, type, PK/FK

CREATE TABLE patient ( patient_id BIGINT NOT NULL, hn VARCHAR(13) NOT NULL, legacy_hn VARCHAR(13) NULL, -- HN เดิมจาก HosXP (migration map) citizen_id VARCHAR(13) NULL, title VARCHAR(20) NULL, first_name VARCHAR(100) NOT NULL, last_name VARCHAR(100) NOT NULL, gender CHAR(1) NOT NULL, -- M / F / U birth_date DATE NULL, phone VARCHAR(20) NULL, allergy_flag BOOLEAN NOT NULL DEFAULT FALSE, created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP, updated_at TIMESTAMP NULL, CONSTRAINT pk_patient PRIMARY KEY (patient_id), CONSTRAINT uq_patient_hn UNIQUE (hn) ); CREATE TABLE visit ( visit_id BIGINT NOT NULL, patient_id BIGINT NOT NULL, visit_no VARCHAR(15) NOT NULL, visit_date TIMESTAMP NOT NULL, sp_code VARCHAR(10) NOT NULL, -- จุดบริการ (9 จุด) doctor_id BIGINT NULL, visit_type VARCHAR(15) NOT NULL, -- OPD / ER status VARCHAR(15) NOT NULL, created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP, CONSTRAINT pk_visit PRIMARY KEY (visit_id), CONSTRAINT uq_visit_no UNIQUE (visit_no), CONSTRAINT fk_visit_patient FOREIGN KEY (patient_id) REFERENCES patient (patient_id), CONSTRAINT fk_visit_doctor FOREIGN KEY (doctor_id) REFERENCES users (user_id) ); CREATE TABLE emar ( emar_id BIGINT NOT NULL, admission_id BIGINT NOT NULL, -- Smart IPD presc_item_id BIGINT NOT NULL, administered_by BIGINT NULL, -- พยาบาลผู้ให้ยา admin_time TIMESTAMP NULL, dose VARCHAR(30) NULL, status VARCHAR(15) NOT NULL, -- given / held / refused created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP, CONSTRAINT pk_emar PRIMARY KEY (emar_id), CONSTRAINT fk_emar_admission FOREIGN KEY (admission_id) REFERENCES admission (admission_id) ON DELETE CASCADE, CONSTRAINT fk_emar_presc_item FOREIGN KEY (presc_item_id) REFERENCES prescription_item (presc_item_id), CONSTRAINT fk_emar_nurse FOREIGN KEY (administered_by) REFERENCES users (user_id) );

แนวทาง Index: สร้าง Index บนทุกคอลัมน์ FK และคอลัมน์ที่ใช้ใน WHERE/JOIN/ORDER BY บ่อย · หลีกเลี่ยง Index เกินจำเป็น (กระทบ INSERT/UPDATE) · พิจารณา composite index ตามรูปแบบคิวรีจริง
